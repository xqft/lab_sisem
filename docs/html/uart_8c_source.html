41void uart_init() {
42 UCA0CTL1 |= UCSWRST; // Set reset
43
44 // Estado por defecto luego de un reset
45 // UCA0CTL0 &= ~UCPEN; // Set paridad off
46 // UCA0CTL0 |= UC7BIT; // Set 8-bit
47
48 UCA0CTL1 |= UCSSEL_1; // Set fuente ACLK
49 UCA0CTL1 |= UCSSEL0; // ""
50
51 // Estado por defecto luego de un reset
52 // BCSCTL1 &= ~XTS; // Modo 0 para el oscilador LFXT1 (selecciona low frequency).
53 // BCSCTL1 |= DIVA_0; // Divisor en /1.
54 // BCSCTL3 |= LFXT1S_0; // Cristal de 32768-Hz para el oscilador LFXT1.
55
56 // Establecer baud-rate 9600 con fuente de 32 kHz.
57 // Ref: Tabla 15-4, Family User's Guide.
58 UCA0BR1 = 0x00;
59 UCA0BR0 = 0x03;
60 UCA0MCTL |= UCBRS_3;
61 UCA0MCTL |= UCBRF_0;
62
63 UCA0CTL1 &= ~UCSWRST; // Reestablecer reset
64 IE2 |= UCA0RXIE; // Habilito interrupciones de recepcion
65}

68 const uint8_t *error_msg = "tx overflow";

99#pragma vector = USCIAB0TX_VECTOR
100__interrupt void tx_isr(void) {
101 if (tx_data_count < tx_data_length) {
102 UCA0TXBUF = tx_data[tx_data_count];
103 tx_data_count++;
104 } else {
105 IE2 &= ~UCA0TXIE;
106 tx_data_count = 0;
107 }
108 __low_power_mode_off_on_exit();

111#pragma vector = USCIAB0RX_VECTOR
112__interrupt void rx_isr(void) {
113 // Siguiente dato disponible del buffer de recepcion
114 static volatile uint8_t rx_data_count = 0;
115
116 char received_char = UCA0RXBUF; // Leer el caracter recibido
117
118 if (received_char == '\r') {
119 if (rx_block_flag) {
120 // Al terminar el mensaje invalido, levanto el bloqueo
121 rx_block_flag = 0;
122 return;
123 }
124
125 rx_data_length = rx_data_count;
126 rx_data_count = 0;
127 *rx_received_flag = 1;
128 } else if (rx_data_count < RX_DATA_MAX_LEN) // saturar en el limite del buffer
129 {
130 if (rx_block_flag) {
131 // Si hay un error, bloqueo la recepcion hasta terminar el mensaje.
132 return;
133 }
134
135 rx_data[rx_data_count] = received_char;
136 rx_data_count++;
137 } else {
138 // Buffer overflow, bloqueo la recepcion y reinicio el buffer
139 rx_data_length = 0;
140 rx_data_count = 0;
141
142 *rx_error_flag = 1;
143 rx_block_flag = 1;
144 }
145 __low_power_mode_off_on_exit();
146}
